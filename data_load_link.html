load instacart data and plot it!

wq2151

10/27/2019

data("instacart")

most_aisle_orders = instacart %>%
  count(aisle, sort = T)
  
most_aisle_orders %>% 
  filter(n > 10000) %>% 
  mutate(aisle = fct_reorder(aisle, -n)) %>% 
  plot_ly(x = ~aisle, y = ~n, type = "bar", color = ~aisle)
